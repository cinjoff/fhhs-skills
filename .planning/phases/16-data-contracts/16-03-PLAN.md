  <name>Task 1: Create bin/lib/paths.cjs</name>
  <files>bin/lib/paths.cjs</files>
  <action>
  Create a module exporting path-building functions for all .planning/ artifacts:

  ```js
  // Phase directory
  exports.phaseDir = (planningDir, phaseId) => path.join(planningDir, 'phases', phaseId);

  // Plan file: {PHASE}-{NN}-PLAN.md
  exports.planPath = (planningDir, phaseId, planNum) =>
    path.join(planningDir, 'phases', phaseId, `${phaseId.split('-')[0]}-${String(planNum).padStart(2,'0')}-PLAN.md`);

  // Summary file: {PHASE}-{NN}-SUMMARY.md
  exports.summaryPath = (planningDir, phaseId, planNum) =>
    path.join(planningDir, 'phases', phaseId, `${phaseId.split('-')[0]}-${String(planNum).padStart(2,'0')}-SUMMARY.md`);

  // Context: {PHASE}-CONTEXT.md (phase number only, not full ID)
  exports.contextPath = (planningDir, phaseId) =>
    path.join(planningDir, 'phases', phaseId, `${phaseId.split('-')[0]}-CONTEXT.md`);

  // Research
  exports.researchPath = (planningDir, phaseId, topic) =>
    path.join(planningDir, 'phases', phaseId, topic ? `${phaseId.split('-')[0]}-RESEARCH-${topic}.md` : `${phaseId.split('-')[0]}-RESEARCH.md`);

  // State files
  exports.statePath = (planningDir) => path.join(planningDir, 'STATE.md');
  exports.roadmapPath = (planningDir) => path.join(planningDir, 'ROADMAP.md');
  exports.projectPath = (planningDir) => path.join(planningDir, 'PROJECT.md');
  exports.requirementsPath = (planningDir) => path.join(planningDir, 'REQUIREMENTS.md');
  exports.decisionsPath = (planningDir) => path.join(planningDir, 'DECISIONS.md');

  // Auto files
  exports.autoStatePath = (planningDir) => path.join(planningDir, '.auto-state.json');
  exports.autoLogPath = (planningDir) => path.join(planningDir, '.auto-log.jsonl');

  // Config
  exports.configPath = (planningDir) => path.join(planningDir, 'config.json');

  // Codebase mapping
  exports.codebasePath = (planningDir, file) => path.join(planningDir, 'codebase', file);

  // Design docs
  exports.designPath = (planningDir, topic) => path.join(planningDir, 'designs', topic);

  // Glob patterns for finding files
  exports.PLAN_GLOB = '*-PLAN.md';
  exports.SUMMARY_GLOB = '*-SUMMARY.md';
  exports.CONTEXT_GLOB = '*-CONTEXT.md';
  ```

  Also export regex patterns for matching:
  ```js
  exports.PLAN_REGEX = /^(\d+(?:\.\d+)*)-(\d+)-PLAN\.md$/;
  exports.SUMMARY_REGEX = /^(\d+(?:\.\d+)*)-(\d+)-SUMMARY\.md$/;
  ```
  </action>

  <done>All .planning/ path patterns are defined in bin/lib/paths.cjs as exported functions</done>
</task>
