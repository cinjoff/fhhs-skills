  1. After the `parseSessionMetrics()` function (~line 579), add an `aggregatePhaseMetrics(stepHistory)` function:
     ```javascript
     function aggregatePhaseMetrics(stepHistory) {
       const phases = {};
       for (const entry of stepHistory) {
         if (!entry.phase) continue;
         if (!phases[entry.phase]) {
           phases[entry.phase] = { tokens_in: 0, tokens_out: 0, cost_estimate: 0, elapsed_ms: 0, steps: 0 };
         }
         const p = phases[entry.phase];
         p.tokens_in += (entry.metrics && entry.metrics.tokens_in) || 0;
         p.tokens_out += (entry.metrics && entry.metrics.tokens_out) || 0;
         p.cost_estimate += entry.cost_estimate || 0;
         p.elapsed_ms += entry.elapsed_ms || 0;
         p.steps++;
       }
       return phases;
     }
     ```

     const phaseIds = Object.keys(phaseCosts);
     if (phaseIds.length > 0) {
       log('');
       log('  Per-phase breakdown:');
       log('  Phase                | Steps | Tokens In | Tokens Out | Cost    | Time');
       log('  ---------------------|-------|-----------|------------|---------|------');
       for (const pid of phaseIds) {
         const pc = phaseCosts[pid];
         const padPhase = pid.padEnd(21);
         const padSteps = String(pc.steps).padStart(5);

     ```

     - `phase`: phase directory name
     - `step`: pipeline step (plan-work | plan-review | build | review)
     - `elapsed`: wall-clock time in ms

     - `reads`: count of Read tool calls (lower = better with ctx_search)
     - `ctx_hits`: count of ctx_search/ctx_batch_execute calls

     ### Per-Phase Cost Aggregation

     The orchestrator aggregates stepHistory into `phase_costs` in `.auto-state.json`:

     ```json
     {
       "phase_costs": {

           "tokens_in": 120000,
           "tokens_out": 35000,
           "cost_estimate": 1.85,
           "elapsed_ms": 1420000,
           "steps": 4
         }
       }
     }
