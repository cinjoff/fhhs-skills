```javascript
function appendSessionHistory(projectDir) {
  const historyPath = path.join(os.homedir(), '.claude', 'tracker', 'history.jsonl');
  const summary = {
    project: path.basename(projectDir),
    project_path: projectDir,
    started_at: _autoStatus.startedAt,
    ended_at: new Date().toISOString(),
    elapsed_ms: Date.now() - new Date(_autoStatus.startedAt).getTime(),
    phases_total: _autoStatus.phasesTotal,
    phases_completed: _autoStatus.phasesCompleted,
    phases_failed: _autoStatus.phasesFailed,
    step_history: _autoStatus.stepHistory,
    step_averages: computeStepAverages(_autoStatus.stepHistory),
  };
  try {
    fs.appendFileSync(historyPath, JSON.stringify(summary) + '\n', 'utf-8');
  } catch { /* non-fatal */ }
}

function computeStepAverages(history) {
  const byStep = {};
  for (const h of history) {
    if (!byStep[h.step]) byStep[h.step] = { count: 0, total_ms: 0, total_turns: 0, total_tokens_in: 0 };
    byStep[h.step].count++;
    byStep[h.step].total_ms += h.elapsed_ms || h.duration_ms || 0;
    byStep[h.step].total_turns += (h.metrics && h.metrics.num_turns) || 0;
    byStep[h.step].total_tokens_in += (h.metrics && h.metrics.tokens_in) || 0;
  }
  const avgs = {};
  for (const [step, data] of Object.entries(byStep)) {
    avgs[step] = {
      avg_ms: Math.round(data.total_ms / data.count),
      avg_turns: Math.round(data.total_turns / data.count),
      avg_tokens_in: Math.round(data.total_tokens_in / data.count),
      count: data.count,
    };
  }
  return avgs;
}
```

Call `appendSessionHistory(projectDir)` right before the final writeAutoStatus with `active: false`.
